# Les conditions

## If
```
var nb = 12;
if (nb < 0) {
  console.log('0');
} else if (nb < 5) {
  console.log('< 5');
} else if (nb > 5 && nb < 20) {
  console.log('nb > 5 && nb < 20');
} else {
  console.log('autre');
}
```

## switch
```
var langue = 'fr';
switch (langue) {
  case 'fr':
    console.log('bonjour');
    break;
  case 'en':
    console.log('hello');
    break;
  default:
    console.log('??');
}
```


# Les boucles

  console.log(i);
}
